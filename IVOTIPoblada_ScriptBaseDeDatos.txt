
CREATE DATABASE ivotidb;

CREATE TABLE CentroVotacion (
    IdCentroVotacion VARCHAR(6) PRIMARY KEY,
    NombreCentroVotacion VARCHAR(255) NOT NULL,
    Departamento VARCHAR(255) NOT NULL,
    Municipio VARCHAR(255) NOT NULL,
    Distrito VARCHAR(255) NOT NULL
);

#Crear la tabla de VOTANTES
CREATE TABLE Votante (
    DUI INT PRIMARY KEY,
    nombresVotante VARCHAR(255) NOT NULL,
    apellidosVotante VARCHAR(255) NOT NULL,
    fechaNacimientoVotante DATE NOT NULL,
    sexoVotante VARCHAR(10) CHECK (sexoVotante IN ('masculino', 'femenino')) NOT NULL,
    celularVotante INT NOT NULL,
    fechaExpedicionDuiVotante DATE NOT NULL,
    IdCentroVotacion VARCHAR(6),
	haVotado BIT DEFAULT 0 NOT NULL,
    FOREIGN KEY (IdCentroVotacion) REFERENCES CentroVotacion(IdCentroVotacion)
);

#Crear la tabla de PARTIDOS POLITICOS
CREATE TABLE Partidos (
    idPartido VARCHAR(4) PRIMARY KEY,
    nombrePartido VARCHAR(60) NOT NULL,
    abreviacionPartido VARCHAR(10) NOT NULL,
    rutaLogo VARCHAR(255) 
);

#Crear la Tabla de los CANDIDATOS
CREATE TABLE Candidatos (
    IdCandidato VARCHAR(6) PRIMARY KEY,
    nombresCandidato VARCHAR(100) NOT NULL,
    apellidosCandidato VARCHAR(100) NOT NULL,
    fotoCandidato VARCHAR(255) NULL,
    idPartido VARCHAR(4) NOT NULL,
    FOREIGN KEY (idPartido) REFERENCES Partidos(idPartido)
);


#Crear la tabla donde se REGISTRARAN los VOTOS
CREATE TABLE Votos (
    IdVoto INT PRIMARY KEY auto_increment,
	horaVoto TIME NOT NULL,
    DUI INT,
    IdCandidato VARCHAR(6),
    IdCentroVotacion VARCHAR(6),
    FOREIGN KEY (DUI) REFERENCES Votante(DUI),
    FOREIGN KEY (IdCandidato) REFERENCES Candidatos(IdCandidato),
    FOREIGN KEY (IdCentroVotacion) REFERENCES CentroVotacion(IdCentroVotacion)
);
